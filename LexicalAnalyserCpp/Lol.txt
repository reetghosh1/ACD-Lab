#include<bits/stdc++.h>
using namespace std;
vector<char>tape;
void state0(int);
void state1(int);
void state2(int);
void state3(int);
void state4(int);
void state5(int);
int main(){
cout<<"Enter Tape:";
string input;
cin>>input;
if(input.length()==0){
    cout<<"String Rejected";
    exit(0);
}
for(int i=0;i<input.length();i++)
    tape.push_back(input[i]);
tape.push_back('B');
state0(0);
}
void state0(int index){
    cout<<"State 0 --> a(Write x) --> ";
    if(tape[index]=='a'){
        tape[index]='x';
        state1(index+1);
    }
    else{
        cout<<"String Rejected";
        exit(0);
    }
}
void state1(int index){
    cout<<"State 1 --> ";
    if(tape[index]=='a'){
        cout<<"a(Write a) --> ";
        state1(index+1);
    }
    else if(tape[index]=='y'){
        cout<<"y(Write y) --> ";
        state1(index+1);
    }
    else if(tape[index]=='b'){
        tape[index]='y';
        cout<<"b(Write y) --> ";
        state2(index-1);
    }
    else if(tape[index]=='B'){
        cout<<"B(Write B) --> ";
        state3(index-1);
    }
    else{
        cout<<"String Rejected";
        exit(0);
    }
}
void state2(int index){
    cout<<"State 2 --> ";
    if(tape[index]=='a'){
        cout<<"a(Write a) --> ";
        state2(index-1);
    }
    else if(tape[index]=='y'){
        cout<<"y(Write y) --> ";
        state2(index-1);
    }
    else if(tape[index]=='x'){
        cout<<"x(Write x) --> ";
        state0(index+1);
    }
    else{
        cout<<"String Rejected";
        exit(0);
    }
}
void state3(int index){
    cout<<"State 3 --> ";
    if(tape[index]=='a'){
        cout<<"a(Write a) --> ";
        state3(index-1);
    }
    else if(tape[index]=='y'){
        cout<<"y(Write y) --> ";
        state3(index-1);
    }
    else if(tape[index]=='x'){
        cout<<"x(Write x) --> ";
        state4(index+1);
    }
    else{
        cout<<"String Rejected";
        exit(0);
    }
}
void state4(int index){
    cout<<"State 4 --> ";
    if(tape[index]=='a'){
        cout<<"a(Write a) --> ";
        state5(index+1);
    }
    else{
        cout<<"String Rejected";
        exit(0);
    }
}
void state5(int index){
    cout<<"State 5 --> ";
    if(tape[index]=='y'){
        cout<<"y(Write y) --> ";
        state5(index+1);
    }
    else if(tape[index]=='B'){
        cout<<"String accepted";
        exit(0);
    }
    else{
        cout<<"String Rejected";
        exit(0);
    }
}
